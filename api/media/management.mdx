## Overview

The Media Management endpoint allows you to retrieve a list of all media files uploaded to your Mediastream account. This includes videos in various states: processing, ready for streaming, or archived.

<Note>
  This endpoint returns metadata about your media files. To upload new files, use the [Upload Media](/api/media/upload) endpoint.
</Note>

## Authentication

This endpoint requires authentication. The backend automatically includes your Mediastream API credentials configured in the environment variables.

## Request

<CodeGroup>

```bash cURL
curl -X GET "https://your-app.com/api/media" \
  -H "Accept: application/json" \
  -H "Authorization: Bearer YOUR_AUTH_TOKEN"
```

## Response

The endpoint returns a list of media objects with detailed metadata.

<ResponseField name="status" type="string">
  The status of the request. Returns `"success"` when media is retrieved successfully.
</ResponseField>

<ResponseField name="data" type="array">
  Array of media objects. Each object contains:
  
  <ResponseField name="_id" type="string">
    Unique identifier for the media file.
    
    Example: `"60f1a2b3c4d5e6f7g8h9i0j1"`
  </ResponseField>
  
  <ResponseField name="title" type="string">
    Display title of the media file. Defaults to the original filename if not set.
    
    Example: `"Product Demo Video"`
  </ResponseField>
  
  <ResponseField name="file_name" type="string">
    Original filename of the uploaded file.
    
    Example: `"product-demo.mp4"`
  </ResponseField>
  
  <ResponseField name="duration" type="number">
    Duration of the video in seconds.
    
    Example: `245.5` (4 minutes 5.5 seconds)
  </ResponseField>
  
  <ResponseField name="size" type="integer">
    File size in bytes.
    
    Example: `52428800` (50MB)
  </ResponseField>
  
  <ResponseField name="status" type="string">
    Current processing status of the media file.
    
    Possible values:
    - `processing` - Video is being transcoded
    - `ready` - Video is ready for streaming
    - `error` - Processing failed
    - `archived` - Video has been archived
  </ResponseField>
  
  <ResponseField name="thumbnail" type="string">
    URL to the video thumbnail image.
    
    Example: `"https://cdn.mdstrm.com/thumbs/abc123.jpg"`
  </ResponseField>
  
  <ResponseField name="url" type="string">
    Streaming URL for the video.
    
    Example: `"https://mdstrm.com/video/abc123def456"`
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    ISO 8601 timestamp when the media was uploaded.
    
    Example: `"2024-03-05T10:30:00Z"`
  </ResponseField>
  
  <ResponseField name="updated_at" type="string">
    ISO 8601 timestamp when the media was last modified.
    
    Example: `"2024-03-05T11:45:00Z"`
  </ResponseField>
  
  <ResponseField name="format" type="string">
    Video format/codec information.
    
    Example: `"mp4"`
  </ResponseField>
  
  <ResponseField name="resolution" type="object">
    Video resolution information.
    
    <ResponseField name="width" type="integer">
      Video width in pixels.
      
      Example: `1920`
    </ResponseField>
    
    <ResponseField name="height" type="integer">
      Video height in pixels.
      
      Example: `1080`
    </ResponseField>
  </ResponseField>
</ResponseField>

## Filtering and Pagination

While the current implementation returns all media, you can filter results client-side:

```javascript
// Filter by status
const readyVideos = media.filter(item => item.status === 'ready')

// Filter by date
const recentVideos = media.filter(item => {
  const uploadDate = new Date(item.created_at)
  const sevenDaysAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000)
  return uploadDate > sevenDaysAgo
})

// Filter by duration
const shortVideos = media.filter(item => item.duration < 300) // Under 5 minutes

// Search by title
const searchResults = media.filter(item => 
  item.title.toLowerCase().includes('tutorial')
)
```

## Display Media Library

Example Vue.js component for displaying media files:

```vue
<script setup lang="ts">
import { ref, onMounted, computed } from 'vue'
import axios from 'axios'

interface Media {
  _id: string
  title: string
  thumbnail: string
  duration: number
  status: string
  created_at: string
}

const mediaFiles = ref<Media[]>([])
const loading = ref(true)
const filter = ref('all')

const fetchMedia = async () => {
  try {
    const response = await axios.get('/api/media')
    mediaFiles.value = response.data.data
  } catch (error) {
    console.error('Failed to fetch media:', error)
  } finally {
    loading.value = false
  }
}

const filteredMedia = computed(() => {
  if (filter.value === 'all') return mediaFiles.value
  return mediaFiles.value.filter(m => m.status === filter.value)
})

const formatDuration = (seconds: number) => {
  const minutes = Math.floor(seconds / 60)
  const remainingSeconds = Math.floor(seconds % 60)
  return `${minutes}:${remainingSeconds.toString().padStart(2, '0')}`
}

onMounted(fetchMedia)
</script>

<template>
  <div class="p-4">
    <div class="mb-4 flex gap-2">
      <button @click="filter = 'all'" :class="{'bg-primary': filter === 'all'}">
        All
      </button>
      <button @click="filter = 'ready'" :class="{'bg-primary': filter === 'ready'}">
        Ready
      </button>
      <button @click="filter = 'processing'" :class="{'bg-primary': filter === 'processing'}">
        Processing
      </button>
    </div>
    
    <div v-if="loading" class="text-center py-8">
      Loading media files...
    </div>
    
    <div v-else class="grid grid-cols-1 md:grid-cols-3 gap-4">
      <div v-for="media in filteredMedia" :key="media._id" 
           class="rounded-lg overflow-hidden shadow-lg bg-white">
        <img :src="media.thumbnail" :alt="media.title" class="w-full aspect-video object-cover" />
        <div class="p-4">
          <h3 class="font-semibold text-lg">{{ media.title }}</h3>
          <div class="flex justify-between mt-2 text-sm text-gray-600">
            <span>{{ formatDuration(media.duration) }}</span>
            <span class="capitalize">{{ media.status }}</span>
          </div>
        </div>
      </div>
    </div>
  </div>
</template>
```

## Implementation

The backend controller proxies requests to Mediastream's API:

```php
<?php

namespace App\Http\Controllers\Web\Media;

use App\Http\Controllers\Controller;
use App\Http\Services\MediastreamService;
use Illuminate\Http\Request;

class MediaController extends Controller
{
    /**
     * Display a listing of all media.
     */
    public function index()
    {
        $response = MediastreamService::request('media', 'get');
        
        return $response->json();
    }
}
```

### Mediastream Service

The service handles authentication and API communication:

```php
<?php

namespace App\Http\Services;

use Illuminate\Support\Facades\Http;
use Illuminate\Http\Client\Response;

class MediastreamService
{
    public static function request(string $endpoint, string $method = 'get', array $data = []): Response
    {
        $baseUrl = rtrim(env('MEDIASTREAM_API_URL'), '/');
        $url = $baseUrl . '/' . ltrim($endpoint, '/');

        $client = Http::withHeaders([
            'X-API-Token' => env('MEDIASTREAM_API_KEY'),
            'Accept' => 'application/json',
        ]);

        switch (strtolower($method)) {
            case 'post':
                return $client->post($url, $data);
            case 'put':
                return $client->put($url, $data);
            case 'delete':
                return $client->delete($url, $data);
            default:
                return $client->get($url, $data);
        }
    }
}
```

<Tip>
  Configure your Mediastream credentials in `.env`:
  ```env
  MEDIASTREAM_API_URL=https://api.mdstrm.com/v1
  MEDIASTREAM_API_KEY=your_api_key_here
  ```
</Tip>

## Error Responses

<ResponseField name="status" type="string">
  Returns `"error"` when the request fails.
</ResponseField>

<ResponseField name="message" type="string">
  Human-readable error message.
</ResponseField>

<ResponseField name="code" type="string">
  Machine-readable error code.
  
  Common error codes:
  - `unauthorized` - Invalid or missing API credentials
  - `forbidden` - Account doesn't have access to media API
  - `rate_limit_exceeded` - Too many requests
  - `server_error` - Mediastream API error
</ResponseField>

## Best Practices

<AccordionGroup>
  <Accordion title="Caching">
    Cache media lists to reduce API calls:
    ```javascript
    const mediaCache = {
      data: null,
      timestamp: 0,
      ttl: 5 * 60 * 1000 // 5 minutes
    }
    
    const getMedia = async (forceRefresh = false) => {
      const now = Date.now()
      
      if (!forceRefresh && mediaCache.data && (now - mediaCache.timestamp) < mediaCache.ttl) {
        return mediaCache.data
      }
      
      const response = await axios.get('/api/media')
      mediaCache.data = response.data.data
      mediaCache.timestamp = now
      
      return mediaCache.data
    }
    ```
  </Accordion>

  <Accordion title="Lazy Loading">
    Load thumbnails only when visible:
    ```vue
    <img :src="media.thumbnail" 
         loading="lazy" 
         :alt="media.title" 
         class="w-full aspect-video" />
    ```
  </Accordion>

  <Accordion title="Error Handling">
    Handle API errors gracefully:
    ```javascript
    const fetchMediaSafely = async () => {
      try {
        const response = await axios.get('/api/media')
        return response.data.data
      } catch (error) {
        if (error.response?.status === 401) {
          console.error('Authentication failed')
          // Redirect to login
        } else if (error.response?.status === 429) {
          console.error('Rate limit exceeded, please wait')
          // Show rate limit message
        } else {
          console.error('Failed to fetch media:', error)
          // Show generic error
        }
        return []
      }
    }
    ```
  </Accordion>

  <Accordion title="Real-time Updates">
    Poll for updates when videos are processing:
    ```javascript
    const pollForUpdates = async () => {
      const response = await axios.get('/api/media')
      const processing = response.data.data.filter(m => m.status === 'processing')
      
      if (processing.length > 0) {
        // Check again in 30 seconds
        setTimeout(pollForUpdates, 30000)
      }
      
      return response.data.data
    }
    ```
  </Accordion>
